###letcode 1:Two Sum, letcode 18:4Sum，类似的还有3Sum,在数组中寻找几个数的值构成特定目标和，在求两个和中第一想到的是暴力破解，但在求职面试中，我们要注意方法的运用，可以使用c++的unordered_map利用哈希表降低复杂度。

class Solution {//给定一个数组和一个目标值，给出两个数值相加和为目标值的下标
public:
vector<int> twoSum(vector<int>& nums, int target) {
//         for(int i = 0; i < nums.size(); i++){//暴力破解 O(n*logn) time//136ms,9.1MB
//             for(int j = i+1; j < nums.size(); j++){

//                     return vector<int>{i,j};

//                 }

    //一次遍历法
    //讲解：map,内部数据结构为红黑树，因此所有元素插入到map里面都会排好序，而且搜索过程为平衡二叉树搜索，因此时间复杂度为O(logN)。unordered _map，内部利用哈希函数能快速的查找到所需元素
    //Runtime: 8 ms
    //Memory Usage: 9.9 MB
    unordered_map<int,int> rec;
    for(int i = 0; i < nums.size(); i++){
        if(rec.find(target-nums[i]) != rec.end()){
            return {rec[target-nums[i]],i};
        }
        rec[nums[i]] = i;
    }




    return{-1,-1};

}
};

同理求最小和，i，j节点处的值为从开始节点到该节点所经过路径的总和，对该节点来说，有两种方案到这，即从左边来，另一个从上边来，我们要选最小的，所以grid[i][j] += min(grid[i-1][j],grid[i][j-1]);

class Solution {
public:
int minPathSum(vector<vector<int>>& grid) {
     int m = grid.size();
     int n = grid[0].size();
    // vector<vector<int>> dp(m,vector<int>(n,0));
    for(int i = 1; i < m; i++){
        grid[i][0] += grid[i-1][0] ;
    }
    for(int i = 1; i < n; i++){
        grid[0][i] += grid[0][i-1];
    }
    for(int i = 1; i < m; i++){
        for(int j = 1; j < n; j++){
            grid[i][j] += min(grid[i-1][j],grid[i][j-1]);
        }
    }

    return grid[m-1][n-1];
 }
};
